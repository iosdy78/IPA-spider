            justify-content: center;
            align-items: center;
            height: 100vh;
            background-color: #1a1a2e;
            font-family: 'Arial', sans-serif;
            overflow: hidden;
            color: white;
        }
        #game-container {
            position: relative;
            box-shadow: 0 0 50px rgba(0,0,0,0.5);
        }
        canvas {
            background-color: #16213e;
            border-bottom: 5px solid #e94560;
            display: block;
            border-radius: 10px;
        }
        .ui {
            position: absolute;
            top: 10px;
            left: 20px;
            font-size: 20px;
            pointer-events: none;
        }
        #instructions {
            position: absolute;
            bottom: 20px;

            text-align: center;
            font-size: 16px;
            opacity: 0.7;

<body>

<div id="game-container">
    <div class="ui">النقاط: <span id="score">0</span> | الأعلى: <span id="highScore">0</span></div>
    <canvas id="gameCanvas"></canvas>
    <div id="instructions">اضغط على الشاشة أو زر المسافة (Space) للقفز!</div>
</div>

<script>
    const canvas = document.getElementById('gameCanvas');
    const ctx = canvas.getContext('2d');
    const scoreElement = document.getElementById('score');
    const highScoreElement = document.getElementById('highScore');

    // إعدادات الشاشة
    canvas.width = 800;
    canvas.height = 400;

    // متغيرات اللعبة
    let score = 0;
    let highScore = localStorage.getItem('highScore') || 0;
    highScoreElement.innerText = highScore;

    let gameActive = true;
    let gameSpeed = 5;
    
    // اللاعب
    const player = {
        x: 50,
        y: canvas.height - 60,
        width: 40,
        height: 40,
        color: '#4db8ff',
        dy: 0,
        jumpForce: 12,
        gravity: 0.6,
        grounded: false
    };

    // العقبات
    let obstacles = [];

    function spawnObstacle() {
        let size = Math.random() * (60 - 20) + 20;
        obstacles.push({
            x: canvas.width,
            y: canvas.height - size,
            width: 30,
            height: size,
            color: '#e94560'
        });
    }

    // التحكم
    function jump() {
        if (player.grounded && gameActive) {
            player.dy = -player.jumpForce;
            player.grounded = false;
        } else if (!gameActive) {
            resetGame();
        }
    }

    window.addEventListener('keydown', (e) => { if (e.code === 'Space') jump(); });
    canvas.addEventListener('touchstart', (e) => { e.preventDefault(); jump(); });
    canvas.addEventListener('mousedown', jump);

    function update() {
        if (!gameActive) return;

        // الجاذبية والقفز
        player.dy += player.gravity;
        player.y += player.dy;

        if (player.y + player.height > canvas.height) {
            player.y = canvas.height - player.height;
            player.dy = 0;
            player.grounded = true;
        }

        // تحريك العقبات
        if (Math.random() < 0.02) {
            if (obstacles.length === 0 || (canvas.width - obstacles[obstacles.length-1].x) > 150) {
                spawnObstacle();
            }
        }

        for (let i = obstacles.length - 1; i >= 0; i--) {
            obstacles[i].x -= gameSpeed;

            // التصادم
            if (
                player.x < obstacles[i].x + obstacles[i].width &&
                player.x + player.width > obstacles[i].x &&
                player.y < obstacles[i].y + obstacles[i].height &&
                player.y + player.height > obstacles[i].y
            ) {
                gameOver();
            }

            // حذف العقبات التي خرجت وزيادة النقاط
            if (obstacles[i].x + obstacles[i].width < 0) {
                obstacles.splice(i, 1);
                score++;
                scoreElement.innerText = score;
                // زيادة السرعة تدريجياً
                gameSpeed += 0.1;
            }
        }
    }

    function draw() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);

        // رسم اللاعب (مع تأثير توهج)
        ctx.shadowBlur = 15;
        ctx.shadowColor = player.color;
        ctx.fillStyle = player.color;
        ctx.fillRect(player.x, player.y, player.width, player.height);

        // رسم العقبات
        ctx.shadowColor = '#e94560';
        for (let obs of obstacles) {
            ctx.fillStyle = obs.color;
            ctx.fillRect(obs.x, obs.y, obs.width, obs.height);
        }
        ctx.shadowBlur = 0;

        if (!gameActive) {
            ctx.fillStyle = "rgba(0,0,0,0.7)";
            ctx.fillRect(0,0,canvas.width, canvas.height);
            ctx.fillStyle = "white";
            ctx.font = "40px Arial";
            ctx.textAlign = "center";
            ctx.fillText("انتهت اللعبة! 💀", canvas.width/2, canvas.height/2);
            ctx.font = "20px Arial";
            ctx.fillText("اضغط للبدء من جديد", canvas.width/2, canvas.height/2 + 40);
        }
    }

    function gameOver() {
        gameActive = false;
        if (score > highScore) {
            highScore = score;
            localStorage.setItem('highScore', highScore);
            highScoreElement.innerText = highScore;
        }
    }

    function resetGame() {
        score = 0;
        scoreElement.innerText = score;
        obstacles = [];
        gameSpeed = 5;
        gameActive = true;
        player.y = canvas.height - 60;
    }

    function gameLoop() {
        update();
        draw();
        requestAnimationFrame(gameLoop);
    }

    gameLoop();
</script>
